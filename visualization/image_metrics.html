# Load input, per subject
fname = '../results/surface_maps/sub-{0}/sub-{0}_{1}_{2}.native.shape.gii'

subj_data = np.zeros((
    len(nib.load(unfolded).get_arrays_from_intent('NIFTI_INTENT_POINTSET')[0].data),
    len(subjects),
    len(hemis),
    len(maps_dict)
))

for s, subject in enumerate(subjects):
    for h, hemi in enumerate(hemis):
        for i, idx in enumerate(maps_dict.keys()):
            if i != 7:
                subj_data[:,s,h,i] = nib.load(
                    fname.format(
                        subject,
                        f"{'tpl-hires_' if idx == 4 else ''}{maps_dict[idx][0]}",
                        hemi
                    )
                ).darrays[0].data
                
avg_data = np.nanmean(subj_data, axis=(2,1))
